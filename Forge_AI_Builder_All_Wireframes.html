Refactored code (bulkified):
APEX — FIXED (bulkified)
trigger AccountTrigger on Account (before insert, before update) {
  // ✓ Collect all Account Ids first
  Set<Id> accountIds = new Set<Id>();
  for (Account acc : Trigger.new) {
    accountIds.add(acc.Id);
  }
  // ✓ Single SOQL query outside the loop
  Map<Id, List<Contact>> contactMap =
    new Map<Id, List<Contact>>();
  for (Contact c : [SELECT Id, AccountId
    FROM Contact WHERE AccountId IN :accountIds]) {
    if (!contactMap.containsKey(c.AccountId)) {
      contactMap.put(c.AccountId, new List<Contact>());
    }
    contactMap.get(c.AccountId).add(c);
  }
  // ✓ O(1) lookup per account — no governor limit risk
  for (Account acc : Trigger.new) {
    List<Contact> contacts = contactMap.get(acc.Id);
  }
}
